    labels = ['1','2','3','4','5'] #
    sizes = [counter_dic[ii + 1] for ii in range(5)] #

    patches,text1,text2 = plt.pie(sizes,
                          explode=explode,
                          labels=labels,
                          colors=colors,
                          autopct = '%3.2f%%', #
                          shadow = False, #
                          startangle = 90, #
                          pctdistance = 0.7) #
    for t in text1:
        t.set_size(12)
    for t in text2:
        t.set_size(12)

    #patchestexts1labeltexts2
